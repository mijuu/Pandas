	input @input;
	if (@input == 0) close;
	if (@input > 5) {
		mes "[驱魔大师 法海]";
		mes "你不能那么贪婪, 阿弥陀佛.";
		close;
	}
	if (Zeny< @input*.@item[3+.@i]) {
		mes "[驱魔大师 法海]";
		mes "钱不能够给每个人带来快乐, 但我们需要它来支援我的庙和我自己. 请, 尝试杀掉魔物或者卖掉他们的掉落物.";
		close;
	}
	set Zeny,Zeny-.@item[3+.@i]*@input;
	getitem 682,@input;
	mes "[驱魔大师 法海]";
	mes "给你东西, 我希望你能够成功完成任务.";
	close;
