comodo,228,159,4	script	发箍姊姊	8_F_GIRL,{
	mes "[发箍姊姊]";
	mes "你好～";
	mes "要不要试着制作发箍呢? 呼呼~";
	next;
	mes "[发箍姊姊]";
	mes "只要带简单的材料来~";
	mes "我会帮你制作, 很炫的发箍给你~";
	mes "选选看吧~";
	next;
	switch (select("十字架发圈:灯泡发圈:条纹发带:蓝色发带")) {
	case 1:
		setarray .@need,2608,7069;
		setarray .@amount,1,500;
		set .@gain,5036;
		break;
	case 2:
		setarray .@need,2233,746;
		setarray .@amount,1,20;
		set .@gain,5034;
		break;
	case 3:
		set .@need,1099;
		set .@amount,1500;
		set .@gain,5049;
		break;
	case 4:
		setarray .@need,2211,978,7003;
		setarray .@amount,1,1,300;
		set .@gain,5052;
		break;
	}
	for(set .@i,0; .@i<getarraysize(.@need); set .@i,.@i+1) {
		if(countitem(.@need[.@i]) < .@amount[.@i]) {
			mes "[发箍姊姊]";
			mes "嗯~ 你选择了" +getitemname(.@gain)+ "啊!";
			mes "需要几种材料, 请你准备一下吧~";
			next;
			mes "[发箍姊姊]";
			switch(@menu) {
			case 1:
				mes "为了制作十字架发圈需要...";
				mes "^3355FF幸运珠链^000000 1 个";
				mes "^3355FF破裂的铠甲一角^000000 500 个";
				break;
			case 2:
				mes "为了制作灯泡发圈需要...";
				mes "^3355FF头箍^000000 1 个";
				mes "^3355FF玻璃珠^000000 20 个";
				break;
			case 3:
				mes "为了制作条纹发带需要...";
				mes "^3355FF老旧犯人服^000000 1500 个";
				break;
			case 4:
				mes "为了制作蓝色发带需要...";
				mes "^3355FF方头巾^000000 1 个";
				mes "^3355FF蓝色染料^000000 1 个";
				mes "^3355FF虎蜥皮^000000 300 个";
				break;
			}
			mes "总共需要这些";
			next;
			mes "[发箍姊姊]";
			mes "带这些材料来的话, 马上帮你做!";
			close;
		}
	}
	mes "[发箍姊姊]";
	mes "嗯~ 你选择了" +getitemname(.@gain)+ "啊!";
	mes "材料带来了吗";
	next;
	mes "[发箍姊姊]";
	mes "注意!";
	mes "如果你的物品被精练过";
	mes "或者插有卡片";
	mes "我一样会收走";
	mes "所以请注意放在仓库里";
	next;
	mes "[发箍姊姊]";
	mes "带这些材料来的话, 马上帮你做~";
	mes "呜呼~ 辛苦了~";
	next;
	if(select("不要:好的")==1) {
		mes "[发箍姊姊]";
		mes "嗯, 怎么拉";
		mes "小孩子早点回家睡觉去～";
		close;
	}
	mes "^4169E1叮叮当当^000000";
	next;
	for(set .@i,0; .@i<getarraysize(.@need); set .@i,.@i+1) {
		delitem .@need[.@i],.@amount[.@i];
	}
	getitem .@gain,1;
	mes "[发箍姊姊]";
	mes "祝贺你!我已经成功的做好了...";
	close;
}
